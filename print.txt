bool isPrime(long long n)
{
    if(n < 2)
        return false;

    for(long long i = 2; i * i <= n; i++)
    {
        if(n % i == 0)
            return false;
    }

    return true;
}

for(int i = 2; i <= n; i++)
{
    if(isPrime(i))
        cout << i << " ";
}


Problem: Prime Factorization
Purpose:
Print all prime factors of a number.
Input:
60
Output:
2 2 3 5

#include <iostream>
using namespace std;

int main()
{
    long long n;
    cin >> n;

    for(long long i = 2; i * i <= n; i++)
    {
        while(n % i == 0)
        {
            cout << i << " ";
            n /= i;
        }
    }

    if(n > 1)
        cout << n;

    return 0;
}
